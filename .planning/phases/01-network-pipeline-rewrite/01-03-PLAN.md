```cpp
void AsioTCPSocket::handleDisconnect(const std::error_code &ec)
{
  if (ec == asio::error::operation_aborted) return; // 主动关闭，不处理

  LOG_WARN("连接断开: %s", ec.message().c_str());

  // 1. 释放所有按键和鼠标按钮（D-10）
  releaseAllKeys();

  // 2. 取消所有定时器
  m_mouseTimer.cancel();
  m_keyboardPollTimer.cancel();

  // 3. 通知上层断连（D-06 桥接）
  m_connected.store(false, std::memory_order_release);
  m_events->addEvent(Event(EventTypes::SocketDisconnected, getEventTarget()));

  // 4. 自动重连（D-11，仅客户端模式）
  if (m_autoReconnect) {
    scheduleReconnect();
  }
}
```

4. **scheduleReconnect() 实现（D-11 指数退避）：**
```cpp
void AsioTCPSocket::scheduleReconnect()
{
  if (m_reconnectDelay == std::chrono::seconds(0)) {
    // 首次立即重试（D-11）
    LOG_DEBUG("立即尝试重连");
    doReconnect();
    return;
  }

  LOG_DEBUG("计划在 %lld 秒后重连", static_cast<long long>(m_reconnectDelay.count()));
  m_reconnectTimer.expires_after(m_reconnectDelay);
  m_reconnectTimer.async_wait(
    asio::bind_executor(m_strand,
      [this, self = shared_from_this()](const std::error_code &ec) {
        if (ec) return; // 被取消
        doReconnect();
      }
    )
  );

  // 指数退避：1s -> 2s -> 4s -> ... -> 30s max
  m_reconnectDelay = std::min(m_reconnectDelay * 2, std::chrono::seconds(30));
}
```

5. **doReconnect() 实现：**
```cpp
void AsioTCPSocket::doReconnect()
{
  m_reconnectDelay = std::chrono::seconds(1); // 首次失败后 1s
  // 关闭旧 socket（如果还在打开状态）
  asio::error_code ignored_ec;
  m_socket.close(ignored_ec);
  // 发起新的异步连接
  connect(m_targetAddress);
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Asio io_context 线程 -> KeyStateTable | 断连时读取 releaseAll()，使用 mutex 保护 |
| Network -> AsioTCPSocket | 断连事件来自 TCP 错误码，不可信但必须正确处理 |
| AsioTCPSocket -> EventQueue | 断连事件通过 addEvent 桥接到上层，EventQueue 负责线程安全分发 |
| ClientApp 重连 -> AsioTCPSocket | 重连尝试可能失败多次，不应导致资源泄漏 |
