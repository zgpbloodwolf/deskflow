   ```cpp
   void AsioTCPSocket::handleDisconnect(const std::error_code &ec)
   {
     if (ec == asio::error::operation_aborted) return; // Active close, don't handle

     LOG_WARN("Connection disconnected: %s", ec.message().c_str());

     // 1. Release all keys and mouse buttons (D-10)
     releaseAllKeys();

     // 2. Cancel all timers
     m_mouseTimer.cancel();
     m_keyboardPollTimer.cancel();

     // 3. Notify upper layer of disconnect (D-06 bridge)
     m_connected.store(false, std::memory_order_release);
     m_events->addEvent(Event(EventTypes::SocketDisconnected, getEventTarget()));

     // 4. Auto-reconnect (D-11, client mode only)
     if (m_autoReconnect) {
       scheduleReconnect();
     }
   }
   ```

3. **releaseAllKeys() implementation:**
   ```cpp
   void AsioTCPSocket::releaseAllKeys()
   {

       LOG_DEBUG("Disconnect release key: key=%d, mask=0x%04x, button=0x%04x", r.key, r.mask, r.button);
       ProtocolUtil::writef(this, kMsgDKeyUp1_0, r.key, r.mask);
     }
   ```

4. **scheduleReconnect() implementation (D-11 exponential backoff):**
   ```cpp
   void AsioTCPSocket::scheduleReconnect()
   {
     if (m_reconnectDelay == std::chrono::seconds(0)) {
       // First retry immediately (D-11)
       LOG_DEBUG("Immediate reconnect attempt");
       doReconnect();
       return;
     }

     LOG_DEBUG("Reconnect scheduled in %lld seconds", static_cast<long long>(m_reconnectDelay.count()));
     m_reconnectTimer.expires_after(m_reconnectDelay);
     m_reconnectTimer.async_wait(
       asio::bind_executor(m_strand,
         [this, self = shared_from_this()](const std::error_code &ec) {
           if (ec) return; // Cancelled
           doReconnect();
         }
       )
     );

     // Exponential backoff: 1s -> 2s -> 4s -> ... -> 30s max
     m_reconnectDelay = std::min(m_reconnectDelay * 2, std::chrono::seconds(30));
   }
   ```
